\section{QUESTION}
\begin{enumerate}
\item Prove the case that the cut-level is greater than zero and the cut-formula is not principal in the second premise of the proof of cut-elimination in $\PC{G3[mic]}$.
\item Prove the case that the cut-level is greater than zero and the cut-formula is introduced by $(\LX)$ and $(\RX)$ of the proof of cut-elimination in $\PC{G3[mi]}$.
\end{enumerate}

\section{ANSWER}

\begin{enumerate}
    \item\ \begin{description}
        \item[in $\PC{G3c}$] Every $(\Cut)$ in $\PC{G3c}$ which satisfying the condition must have the following form:
        \begin{prooftree*}
        \hypo{}
        \ellipsis{$\delta_0$}{\vdash_a\GG\TO\D,\A}
        \hypo{}
        \ellipsis{$\delta_{10}$}{\vdash_{b-1}\A,\T\TO\LL}
        \hypo{}
        \ellipsis{$\delta_{11}$}{\vdash_{b-1}\A,\SM\TO\PP}
        \delims{ \left[ }{ \right] }
        \infer2[J]{\vdash_b\A,\GG\TO\D}
        \infer2[$\Cut^{\A}_{a+b}$]{\vdash\GG\TO\D}
        \end{prooftree*}
        
        We can transform into the following proof:
        \begin{prooftree*}
        \hypo{}
        \ellipsis{$\delta_0$}{\vdash_a\GG\TO\D,\A}
        \infer1[W]{\vdash_a\GG,\T\TO\D,\LL,\A}
        \hypo{}
        \ellipsis{$\delta_{10}$}{\vdash_{b-1}\A,\T\TO\LL}
        \infer1[W]{\vdash_{b-1}\A,\GG,\T\TO\D,\LL}
        \infer2[$\Cut^{\A}_{a+b-1}$]{\GG,\T\TO\D,\LL}
        \hypo{}
        \ellipsis{$\delta_0$}{\vdash_a\GG\TO\D,\A}
        \infer1[W]{\vdash_a\GG,\SM\TO\D,\PP,\A}
        \hypo{}
        \ellipsis{$\delta_{11}$}{\vdash_{b-1}\A,\SM\TO\PP}
        \infer1[W]{\vdash_{b-1}\A,\GG,\SM\TO\D,\PP}
        \infer2[$\Cut^{\A}_{a+b-1}$]{\GG,\SM\TO\D,\PP}
        \delims{ \left[ }{ \right] }
        \infer2[J]{\GG,\GG\TO\D,\D}
        \infer1[C]{\GG\TO\D}
        \end{prooftree*}
        \item[in $\PC{G3[mi]}$]  Every $(\Cut)$ in $\PC{G3[mi]}$ which satisfying the condition must have the following form:
        \begin{prooftree*}
        \hypo{}
        \ellipsis{$\delta_0$}{\vdash_a\GG\TO\A}
        \hypo{}
        \ellipsis{$\delta_{10}$}{\vdash_{b-1}\A,\T\TO(\theta)}
        \hypo{}
        \ellipsis{$\delta_{11}$}{\vdash_{b-1}\A,\SM\TO(\eta)}
        \delims{ \left[ }{ \right] }
        \infer2[J]{\vdash_b\A,\GG\TO(\delta)}
        \infer2[$\Cut^{\A}_{a+b}$]{\vdash\GG\TO(\delta)}
        \end{prooftree*}
        
        We can transform into the following proof:
        \begin{prooftree*}
        \hypo{}
        \ellipsis{$\delta_0$}{\vdash_a\GG\TO\A}
        \infer1[$\LW$]{\vdash_a\GG,\T\TO\A}
        \hypo{}
        \ellipsis{$\delta_{10}$}{\vdash_{b-1}\A,\T\TO(\theta)}
        \infer1[LW]{\vdash_{b-1}\A,\GG,\T\TO(\theta)}
        \infer2[$\Cut^{\A}_{a+b-1}$]{\GG,\T\TO(\theta)}
        \hypo{}
        \ellipsis{$\delta_0$}{\vdash_a\GG\TO\A}
        \infer1[LW]{\vdash_a\GG,\SM\TO\A}
        \hypo{}
        \ellipsis{$\delta_{11}$}{\vdash_{b-1}\A,\SM\TO(\eta)}
        \infer1[LW]{\vdash_{b-1}\A,\GG,\SM\TO(\eta)}
        \infer2[$\Cut^{\A}_{a+b-1}$]{\GG,\SM\TO(\eta)}
        \delims{ \left[ }{ \right] }
        \infer2[J]{\GG,\GG\TO(\delta)}
        \infer1[LC]{\GG\TO(\delta)}
        \end{prooftree*}
    \end{description}
    \item Every $(\Cut)$ in $\PC{G3[mi]}$ which satisfying the condition must have the following form:
    \begin{prooftree*}
    \hypo{}
    \ellipsis{$\delta_{00}$}{\vdash_{a-1}\GG\TO\A^x_t}
    \infer1[$\RX$]{\vdash_a\GG\TO\exists x\A}
    \hypo{}
    \ellipsis{$\delta_{10}$}{\vdash_{b-1}\A^x_y,\GG\TO(\delta)}
    \infer1[$\LX$]{\vdash_b\exists x\A,\GG\TO(\delta)}
    \infer2[$\Cut^{\exists x\A}_{a+b}$]{\GG\TO(\delta)}
    \end{prooftree*}
    
    By lemma 3.5.2($\beta$), we can obtain $\vdash_{b-1}[(\A^x_y)^y_t,\GG^y_t\TO(\delta^y_t)]^*$. And this is exactly $\vdash_{b-1}[\A^x_t,\GG\TO(\delta)]^*$ since $y$ is free in $\GG,\exists x\A(,\delta)$.
    
    Then we can transform the original proof to the following one:
    \begin{prooftree*}
    \hypo{}
    \ellipsis{$\delta_{00},3.5.2(\alpha)$}{\vdash_{a-1}[\GG\TO\A^x_t]^*}
    \hypo{}
    \ellipsis{$\delta_{10},3.5.2(\beta)$}{\vdash_{b-1}[\A^x_t,\GG\TO(\delta)]^*}
    \infer2[$\Cut^{\A^x_t}_{a+b-2}$]{[\GG\TO(\delta)]^{**}}
    \end{prooftree*}
\end{enumerate}
